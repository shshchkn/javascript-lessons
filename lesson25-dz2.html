Второе домашнее задание к 25-му уроку.

Написать функцию, которая бы запрашивала у пользователя число
и выводила бы на экран куб, указанного пользователем числа.

function cube () {
  var askNum = prompt('Введите число и я возведу его в 3-степень!')
  var result = askNum * askNum * askNum;
  document.write('Куб числа ' + askNum + ' равен ' + result);
};

cube();
