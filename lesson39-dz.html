Домашнее задание к 39-му уроку.

Написать функцию, которая получает в качестве параметра массив,
затем перебирает все его элементы и выводит на экран, какая
информация содержится в каждом элементе: число или нечисло.
Массив для примера: var randSimbol = ["Молоко", 77, 11, "Мед", -88];

var randSimbol = ["Молоко", 77, 11, "Мед", -88];

function myModule(array) { // ф-ция принимает параметр массива
  for (var k = 0; k < array.length; k++) { // счетчик перебирает элементы массива
    if (isNaN(array[k])) { // проверяем каждый элемент число ли это
      document.write('Элемент массива ' + array[k] + ' - НЕ число!< br >');
    } else {
      document.write('Элемент массива ' + array[k] + ' - число!< br >');
    }
  }
}

myModule(randSimbol); // вызываем ф-цию, передав ей в качестве параметра наш массив
