Домашнее задание к 42-му уроку.

Ваша задача написать функцию, которая принимает два параметра.
1. Массив, состоящий из чисел с дробями.
2. Метод округления "round", "ceil" или "floor".

На выходе функция должна выводить этот же массив, но при этом все элементы массива,
должны быть округлены с помощью указанного во втором параметре метода объекта Math.

Исходный массив:
var numberArray = [34.82, 12.90, 17.01, 78.51];

function newArray(anyArray, anyMath) { // ф-ция принимает 2 параметра: массив и метод
  for (var k = 0; k < anyArray.length; k++) { // перебираем массив
    if (anyMath == 'round') { // условие для метода
      anyArray[k] = Math.round(anyArray[k]);
      document.write(anyArray[k] + '< br >');
    } else if (anyMath == 'ceil') { // условие для метода
      anyArray[k] = Math.ceil(anyArray[k]);
      document.write(anyArray[k] + '< br >');
    } else if (anyMath == 'floor') { // условие для метода
      anyArray[k] = Math.floor(anyArray[k]);
      document.write(anyArray[k] + '< br >');
    } else {
      document.write('Передан неверный параметр!< br >');
    }
  }
}

newArray(numberArray, 'round'); // вызов ф-ции
