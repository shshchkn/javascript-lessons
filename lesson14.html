К списку

var students = ['Григорий Щепкин', 'Тамара Андреева', 'Василий Уткин', 'Петр Смирнов', 'Алексей Петров'];

students.splice(3, 2); // доходит до id=3 и удаляет 2 элемента

document.write(students); // Григорий Щепкин,Тамара Андреева,Василий Уткин

students.splice(2, 0, 'Виктор Логинов', 'Денис Шаталин'); // доходит до id=2, ничего не удаляет, т.к. 2-ой параметр '0' и добавляет перед ним, что прописано дальше

document.write(students); // Григорий Щепкин,Тамара Андреева,Виктор Логинов,Денис Шаталин,Василий Уткин

students.splice(1, 2, 'Валерий Баринов', 'Никита Михалков'); // доходит до id=1, удаляет 2 элемента и заменяет их на прописанное дальше

document.write(students); // Григорий Щепкин,Валерий Баринов,Никита Михалков,Денис Шаталин,Василий Уткин
