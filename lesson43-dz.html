Домашнее задание к 43-му уроку.

Вам нужно создать функцию, которая будет принимать три параметра.
1. Массив, состоящий из 10 пустых элементов.
2. Начальное значение интервала
3. Конечное значение интервала.

Функция должна заполнить пустой массив случайными числами в указанном диапазоне и вывести
содержимое массива на экран.

Исходный массив для примера:

var randomArray = new Array(10); // пустой массив из 10 элементов

function myRandom (from, to) { // ф-ция рандомной генерации чисел в определенном диапазоне
  return Math.floor((Math.random() * (to - from + 1)) + from);
}

function myModule(anyArray, min, max) { // ф-ция принимает 3 параметра: массив, начало и конец диапазона
  for (var k = 0; k < anyArray.length; k++) { // перебираем элементы массива
    anyArray[k] = myRandom(min, max); // присваиваем каждому элементу случайное число из нашего диапазона с помощью ф-ции генерации сверху
    document.write(anyArray[k] + '< br >');
  }
}

var start = 1,
  finish = 99; // заносим значения диапазона в переменные

document.write('В массив помещаются числа от ' + start + ' до ' + finish + ':< br >');

myModule(randomArray, start, finish); // вызываем ф-цию, передав 3 параметра
