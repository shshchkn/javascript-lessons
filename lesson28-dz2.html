Второе домашнее задание к 28-му уроку.

1. Напишите функцию, которая в качестве параметра получает строку,
а как результат своей работы вычисляет, в строке больше 10 символов или меньше.

2. Запросите у пользователя ввод любого слова и поместите этот ввод в переменную.

3. Вызовите вашу функцию, а в качестве параметра передайте то самое слово,
которое вы сохранили в переменную.

4. Напечатайте на экран сообщение о том, больше 10 символов в этом слове, меньше или равно 10.

function strLength(myString) { // передаем параметр строки
  if (myString.length < 10) { // проверяем условие по кол-ву знаков
    var result = 'В слове меньше 10 знаков';
  } else if (myString.length > 10) {
    var result = 'В слове больше 10 знаков'
  } else {
    var result = 'В слове ровно 10 знаков'
  }
  return result; // возвращаем результат
}

var askUser = prompt('Введите слово?'); // запрашиваем слово

var ansUser = strLength(askUser); // записываем вызов функции в переменную

document.write(ansUser); // выводим сообщение на экран
